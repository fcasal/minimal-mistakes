$$
\begin{align}
{c_1}^x \cdot {c_2}^y &= m^{x\cdot e_1}\cdot m^{y\cdot e_2} \\
				  &= m^{x \cdot e_1 + y\cdot e_2} \\
				  &= m \enspace .
\end{align}
$$
